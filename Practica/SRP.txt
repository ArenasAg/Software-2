1. Principio de Responsabilidad Única (SRP - Single Responsibility Principle)
Descripción:
Una clase debe tener una única razón para cambiar, lo que significa que debe tener una sola responsabilidad o propósito.


// INCORRECTO: Una clase que maneja tanto el almacenamiento como la validación de usuarios
public class User {
    public void saveUser() {
        // Lógica para almacenar el usuario en la base de datos
    }
    
    public boolean validateUser() {
        // Lógica para validar los datos del usuario
        return true;
    }
}


// CORRECTO: Se separa la lógica de validación y almacenamiento en clases diferentes
public class User {
    // Datos del usuario
}

public class UserValidator {
    public boolean validateUser(User user) {
        // Lógica para validar datos del usuario
        return true;
    }
}

public class UserRepository {
    public void saveUser(User user) {
        // Lógica para almacenar el usuario en la base de datos
    }
}
